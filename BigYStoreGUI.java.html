91       // Code for calculations
92       private void placeOrder() {
93           String item = (String) itemComboBox.getSelectedItem();
94           String size = (String) sizeComboBox.getSelectedItem();
95           int toppings = (int) toppingsSpinner.getValue();
96   
97           double basePrice = 0;
98           switch (size) {
99               case "Small":
100                  basePrice = 5;
101                  break;
102              case "Medium":
103                  basePrice = 10;
104                  break;
105              case "Large":
106                  basePrice = 15;
107                  break;
108              case "Super":
109                  basePrice = 20;
110                  break;
111          }
112  
113          double additionalCost = toppings * 0.5;
114  
115          double totalCost = basePrice + additionalCost;
116  
117          outputTextArea.setText("Item: " + item + "\nSize: " + size + "\nToppings: " + toppings +
118                  "\nTotal Cost: $" + totalCost);
119      }

130      // Custom ComboBox renderer to display icons
131      private class CustomComboBoxRenderer extends DefaultListCellRenderer {
132          private ImageIcon icon;
133  
134          public CustomComboBoxRenderer(ImageIcon icon) {
135              this.icon = icon;
136          }
137  
138          @Override
139          public Component getListCellRendererComponent(JList<?> list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
140              super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
141              if (icon != null) {
142                  setIcon(icon);
143              }
144              return this;
145          }
146      }
