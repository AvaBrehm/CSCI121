import java.util.Scanner;


import java.util.Scanner;

public class MoneyConverter {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter your amount in dollars: $");
        double amount = scanner.nextDouble();

        // Calculate the number of quarters, dimes, nickels, and pennies
        int quarters = (int) (amount / 0.25);
        amount %= 0.25;
        int dimes = (int) (amount / 0.10);
        amount %= 0.10;
        int nickels = (int) (amount / 0.05);
        amount %= 0.05;
        int pennies = (int) (amount / 0.01);


        System.out.println("Calculating...");

        try {
            Thread.sleep(1000);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }

        System.out.println(quarters + " quarters");
        System.out.println(dimes + " dimes");
        System.out.println(nickels + " nickels");
        System.out.println(pennies + " pennies");


        scanner.close();
    }
}
